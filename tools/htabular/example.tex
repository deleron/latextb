\newcommand{\gen}[1]{
  \begin{figure}
    \begin{htabular}[#1]
	%% Headers
	% leave the first cell empty
	  \& ~
	  \& \h{Phenomenon A}
	  \& \h{Long Phenomenon B}
	  \& \h{Long Phenomenon C}
	  \& \h{Phenomenon D}
	  \& ~
	  \& \h{Another Phenomenon E}
	  \& \h{Another Phenomenon F}
	  \& \h{Another Phenomenon G}
	  \& ~
	  \& \h{Bonus}
	  \\
	%% Fill the tabular
	\la{Entry A}    	\&~\& \y \& \y \& \y \& \m \&~\& \m \& \m \& \n \&~\& \m\\
	\la{Entry B}    	\&~\& \y \& \y \& \y \& \m \&~\& \m \& \m \& \m \&~\& \m\\
	\la{Entry C}    	\&~\& \y \& \n \& \y \& \m \&~\& \m \& \m \& \y \&~\& \m\\
	\la{Long Entry D}	\&~\& \y \& \y \& \n \& \m \&~\& \y \& \m \& \n \&~\& \y\\
	\la{Long Entry E}	\&~\& \y \& \y \& \y \& \m \&~\& \m \& \m \& \n \&~\& \m\\
    \end{htabular}
    \caption{Options : [#1]}
  \end{figure}
}
